        <script type="text/javascript">

            function log(message) {
                console.log(message);
                var elem = document.getElementById("logOutput");
                elem.value += message + '\r\n';
                elem.scrollTop = elem.scrollHeight;
            }

            function reset() {
                clearLog();
            }

            function clearLog() {
                document.getElementById("logOutput").value = "";
            }

            function showHide(btn) {
                var elem = document.getElementById("logPane");
                if(elem.classList.contains("logPaneExpanded")) {
                    btn.textContent = "Show";
                    elem.classList.remove('logPaneExpanded');
                    elem.classList.add('logPaneCollapsed');
                    var elem = document.getElementById("content");
                    elem.classList.remove('contentExpanded');
                    elem.classList.add('contentCollapsed');
                    var elem = document.getElementById("divTxtAreaBackground");
                    elem.classList.toggle('footerCollapsed');
                    var elem = document.getElementById("divLogInputBackground");
                    elem.classList.toggle('footerCollapsed');
                    var elem = document.getElementById("divLogPaneFooter");
                    elem.classList.toggle('footerCollapsed');
                } else {
                    btn.textContent = "Hide";
                    elem.classList.remove('logPaneCollapsed');
                    elem.classList.add('logPaneExpanded');
                    var elem = document.getElementById("content");
                    elem.classList.remove('contentCollapsed');
                    elem.classList.add('contentExpanded');
                    var elem = document.getElementById("divTxtAreaBackground");
                    elem.classList.toggle('footerCollapsed');
                    var elem = document.getElementById("divLogInputBackground");
                    elem.classList.toggle('footerCollapsed');
                    var elem = document.getElementById("divLogPaneFooter");
                    elem.classList.toggle('footerCollapsed');
                }
            }

            function inputCommand() {
                if (event.keyCode == 13) {
                    var elem = document.getElementById("logInput");
                    if (!elem.value.includes("=") && !elem.value.includes(")")) {
                        eval(elem.value + "()");
                    } else {
                        eval(elem.value);
                    }
                    elem.value = "";
                }
            }

        </script>
